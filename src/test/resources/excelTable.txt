			function hkd1003a03Complete(data){
				if(data.result == 'error' || data.returnCode != '0'){
					showMsg(data.message);
					$("#stockTable2 tbody").html("<tr><td class='no_data' colspan='10'><span>"+data.message+"</span></td></tr>");
					return;
				}

				var str = "";
				var frc_cnt = 0;

				if(typeof(data.grid01) != "undefined" && data.grid01.length > 0){
					for(var i=0; i<data.grid01.length; i++){
						var obj				= data.grid01[i];
						var admn_acno	= converAccountNo(obj.admn_acno);					//운영계좌번호
						var curr_cd		= obj.curr_cd;													//통화코드 (KRW:원화, 그 외:외화)
						var itm_tp_nm	= obj.itm_tp_nm; 												//종목구분명
						var itm_no		= obj.itm_no;														//종목번호
						var itm_nm1		= obj.itm_nm1;													//종목명1
						var crpr			= numberFormat(obj.crpr, "###,###");				//현재가2
						var hldg_q		= numberFormat(obj.hldg_q, "###,###");			//보유수량
						var pchs_a		= numberFormat(obj.pchs_a, "###,###");			//매입금액 (원화)
						var pchs_a1		= numberFormat(obj.pchs_a1, "###,###.00");	//매입금액 (외화 long)
						var ea				= numberFormat(obj.ea, "###,###");				//평가금액 (원화)
						var pitm_ea		= numberFormat(obj.pitm_ea, "###,###.00");	//평가금액 (외화 long)
						// 손익율이 "-"로 넘어오는 경우, 중간에 공백이 존재하는 부분에 대한 처리
						var	ev_pl_a		= obj.ev_pl_a;
						if (ev_pl_a.substring(0,1) == "-") {
							ev_pl_a = "-" + $.trim(ev_pl_a.substring(1));
						}
						ev_pl_a				= updownTagMP(numberFormat(ev_pl_a, "###,###"));	//평가손익 (원화)
						var	itm_ev_pl	= obj.itm_ev_pl;
						if (itm_ev_pl.substring(0,1) == "-") {
							itm_ev_pl = "-" + $.trim(itm_ev_pl.substring(1));
						}
						itm_ev_pl			= updownTagMP(numberFormat(itm_ev_pl, "###,###.00"));	//평가손익 (외화 long)
						var	ev_pl_r		= obj.ev_pl_r;
						if (ev_pl_r.substring(0,1) == "-") {
							ev_pl_r = "-" + $.trim(ev_pl_r.substring(1));
						}
						ev_pl_r				= updownTagMPM(numberFormat(ev_pl_r, "##.##"),"%");	//평가손익률

						var stat = (curr_cd == 'KRW') ? 'krw':'frc';

						str += "<td class=''>"+itm_tp_nm+"</td>";									//종목구분명

				}

				}else{

				}
